'''
randomly split the images into test, train, valid datasets
'''
def my_test_train_valid_splitter(img_folder, seed):
    full_n = len(os.listdir(img_folder))
    full_set = list(range(1,full_n+1))
    # set seed for reproducability
    random.seed(seed)
    # 80% of images go into training
    train_valid = random.sample(list(range(1,full_n+1)),int(.8*full_n))
    # remaining 20% go into test
    test = list(set(full_set)-set(train_valid))
    # reset seed for reproducibility
    random.seed(seed)
    # 80% of train_valid goes to train, remaining 20% goes to valid
    train = random.sample(train_valid, int(.8*len(train_valid)))
    valid = list(set(train_valid)-set(train))
    
    return (train,valid,test)

'''
query the keras tuner trial history for the best objective value
'''
def get_max_objective_for_trials(tuner, objective, index=0):
    #for trial in tuner.oracle.get_best_trials(num_trials=index+1)[index]:
    trial = tuner.oracle.get_best_trials(num_trials=index+1)[index]
    metrics_tracker = trial.metrics 
    metric_histories = metrics_tracker.metrics
    history = metric_histories[objective]
    objective_val = history.get_history()
    max_objective = max([o.value[0] for o in objective_val])
    #print(f'max {objective} for tuner: {max_objective}')
    return max_objective

'''
get the hyperparameters from the tuner
'''
def get_tuner_model_info(tuner, tuner_type, max_models=20):
    #[model_type, preprocess_rescale, preprocess_random_flip, batch_norm, dense_units, dropout_rate, learning_rate, tuner/epochs, tuner/initial_epoch, tuner/bracket, tuner/round]
    results = [] 
    models = tuner.get_best_models(num_models=max_models)
    hyperparams = tuner.get_best_hyperparameters(num_trials=max_models)
    for i in range(len(models)):
        
        _hyperparams = hyperparams[i]
        val_acc = get_max_objective_for_trials(tuner, 'val_acc', i)

        stats = [
            _hyperparams.get('model_type'),
            _hyperparams.get('preprocess_rescale'),
            _hyperparams.get('preprocess_random_flip'),
            _hyperparams.get('batch_norm'),
            _hyperparams.get('dense_units'),
            _hyperparams.get('dropout_rate'),
            _hyperparams.get('learning_rate'),
            tuner_type,
            val_acc
        ]
        results.append(stats)

    return results

class WasteClassifierHyperModel(kt.HyperModel):

    def build(self, hp):
        model_type = hp.Choice("model_type", MODEL_TYPES)
        
        model = K.Sequential()

        if hp.Boolean("preprocess_rescale"):
            model.add(K.layers.Rescaling(1./255))

        # vgg, densenet, resnet50 require 224x224 images, base cnn does not
        if model_type in ['vgg16', 'densenet121','resnet50']:
            img_height, img_width = 224, 224
        else:
            img_height, img_width = 300, 300
            
            
        input_shape = (img_height, img_width, IMG_DIMENSIONS)
        model.add(K.layers.Resizing(img_height, img_width))
        if hp.Boolean("preprocess_random_flip"):
            model.add(K.layers.RandomFlip())

        #model.add(K.layers.CategoryEncoding(num_tokens=num_classes, output_mode="one_hot"))

        if model_type == 'basecnn':
            model = add_basecnn_layers_to_model(hp, model, input_shape)
        elif model_type == 'densenet121':
            model = add_densenet121_layers_to_model(hp, model, input_shape)
        elif model_type == 'resnet50':
            model = add_resnet50_layers_to_model(hp, model, input_shape)
        elif model_type == 'vgg16':
            model = add_vgg16_layers_to_model(hp, model, input_shape)
        else:
            raise Exception(f'model_type: {model_type} is not supported!')

        # flatted input through 3 dense layers
        model.add(K.layers.Flatten())

        if hp.Boolean("batch_norm"):
            model.add(K.layers.BatchNormalization())

        model.add(K.layers.Dense(hp.Int('dense_units', min_value=32, max_value=256, step=32, default=128), activation='relu'))
        model.add(K.layers.Dropout(rate=hp.Float('dropout_rate',min_value=0.0,max_value=0.5,default=0.2,step=0.1)))
        
        # add a fully connected layer for the final classification
        model.add(K.layers.Dense(num_classes, activation='softmax'))

        #hp_learning_rate = hp.Float('learning_rate',min_value=1e-5,max_value=1e-2,sampling='LOG',default=1e-3)
        hp_learning_rate = hp.Choice('learning_rate', values = TUNING_LEARNING_RATES)
        model.compile(
            optimizer=tf.keras.optimizers.Adam(learning_rate=hp_learning_rate),
            loss='categorical_crossentropy',
            metrics=['acc']
        )
        
        return model
